\chapter{ESP32 Firmware Source Code}

The complete Arduino firmware for the ESP32 sensor and actuator node is listed below.

\begin{lstlisting}[style=mystyle, language=C++, caption={ESP32.ino --- complete firmware source code}, label={lst:esp32}]
// ================================================================
// ESP32 - Smart Home for Elderly Care
// Project: IOT-SMARTHOME-ELDERLY
//
// MQTT Topics:
//   Subscribe : home/control/#        <- command from backend
//   Publish   : home/status/{code}    -> device state feedback
//   Publish   : home/sensors/{code}   -> real-time sensor data
//
// Device codes (match `code` column in MySQL `device` table):
//   den   -> light relay (Relay 1)
//   quat  -> fan relay   (Relay 2)
// ================================================================

#include <WiFi.h>
#include <WiFiClientSecure.h>
#include <PubSubClient.h>
#include <DHT.h>

// WiFi credentials
const char* ssid         = "YOUR_SSID";
const char* password     = "YOUR_WIFI_PASSWORD";

// EMQX Cloud MQTT Broker
const char* mqttServer   = "b2652cf8.ala.us-east-1.emqxsl.com";
const int   MQTT_PORT    = 8883;
const char* mqttUser     = "YOUR_MQTT_USER";
const char* mqttPassword = "YOUR_MQTT_PASSWORD";

// Pin Mapping
#define DHT_PIN          15
#define DHTTYPE          DHT11
#define LIGHT_SENSOR_PIN 32
#define RELAY_DEN_PIN    17
#define RELAY_QUAT_PIN   16
#define PIR_SENSOR_PIN   27
#define BUTTON_DEN_PIN   26
#define BUTTON_QUAT_PIN  25
#define SENSOR_INTERVAL  5000   // ms

DHT              dht(DHT_PIN, DHTTYPE);
WiFiClientSecure espClient;
PubSubClient     client(espClient);
unsigned long    lastReconnectAttempt = 0;

void publishStatus(const char* code, const char* payload) {
  if (!client.connected()) return;
  char topic[64];
  snprintf(topic, sizeof(topic), "home/status/%s", code);
  client.publish(topic, payload, true);
}

void setRelay(uint8_t pin, const char* code, bool on) {
  digitalWrite(pin, on ? HIGH : LOW);
  publishStatus(code, on ? "ON" : "OFF");
}

void mqttCallback(char* topic, byte* message, unsigned int length) {
  String msg;
  for (unsigned int i = 0; i < length; i++) msg += (char)message[i];
  msg.trim();
  String devCode = String(topic);
  devCode = devCode.substring(devCode.lastIndexOf('/') + 1);
  devCode.toLowerCase();
  bool on = msg.equalsIgnoreCase("ON");
  if      (devCode == "den")  setRelay(RELAY_DEN_PIN,  "den",  on);
  else if (devCode == "quat") setRelay(RELAY_QUAT_PIN, "quat", on);
}

void reconnect() {
  String id = "ESP32_" + WiFi.macAddress();
  if (client.connect(id.c_str(), mqttUser, mqttPassword)) {
    client.subscribe("home/control/#");
    publishStatus("den",  "OFF");
    publishStatus("quat", "OFF");
  }
}

void setupWiFi() {
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) delay(500);
}

void handleButtons() {
  static bool btnD = HIGH, btnQ = HIGH;
  static unsigned long tD = 0, tQ = 0;
  bool cD = digitalRead(BUTTON_DEN_PIN);
  bool cQ = digitalRead(BUTTON_QUAT_PIN);
  if (cD == LOW && btnD == HIGH && millis() - tD > 50) {
    tD = millis();
    setRelay(RELAY_DEN_PIN, "den", !digitalRead(RELAY_DEN_PIN));
  }
  if (cQ == LOW && btnQ == HIGH && millis() - tQ > 50) {
    tQ = millis();
    setRelay(RELAY_QUAT_PIN, "quat", !digitalRead(RELAY_QUAT_PIN));
  }
  btnD = cD; btnQ = cQ;
}

void handleSensors(unsigned long now) {
  static unsigned long lastSend = 0;
  if (now - lastSend >= SENSOR_INTERVAL) {
    lastSend = now;
    float temp  = dht.readTemperature();
    float humid = dht.readHumidity();
    int   light = map(analogRead(LIGHT_SENSOR_PIN), 0, 4095, 0, 100);
    if (!client.connected()) return;
    if (!isnan(temp))
      client.publish("home/sensors/temperature",
                     String(temp, 1).c_str(), true);
    if (!isnan(humid))
      client.publish("home/sensors/humidity",
                     String(humid, 1).c_str(), true);
    client.publish("home/sensors/light", String(light).c_str(), true);
  }
  static String lastPir = "";
  String curPir = digitalRead(PIR_SENSOR_PIN) == HIGH ?
                  "DETECTED" : "CLEAR";
  if (curPir != lastPir) {
    lastPir = curPir;
    if (client.connected())
      client.publish("home/sensors/pir", curPir.c_str(), true);
  }
}

void setup() {
  Serial.begin(115200);
  pinMode(RELAY_DEN_PIN,   OUTPUT); digitalWrite(RELAY_DEN_PIN,  LOW);
  pinMode(RELAY_QUAT_PIN,  OUTPUT); digitalWrite(RELAY_QUAT_PIN, LOW);
  pinMode(PIR_SENSOR_PIN,  INPUT);
  pinMode(BUTTON_DEN_PIN,  INPUT);
  pinMode(BUTTON_QUAT_PIN, INPUT);
  dht.begin();
  setupWiFi();
  if (MQTT_USE_CA_CERT && strlen(MQTT_ROOT_CA) > 0) {
    espClient.setCACert(MQTT_ROOT_CA);
  } else if (MQTT_ALLOW_INSECURE_TLS) {
    espClient.setInsecure();
  }
  client.setServer(mqttServer, MQTT_PORT);
  client.setCallback(mqttCallback);
}

void loop() {
  unsigned long now = millis();
  if (!client.connected()) {
    if (now - lastReconnectAttempt > 5000) {
      lastReconnectAttempt = now;
      reconnect();
    }
  } else {
    client.loop();
  }
  handleSensors(now);
  handleButtons();
}
\end{lstlisting}

from datetime import datetime
from app.domain.entities import SensorReading
from app.infrastructure.database import db
from app.extensions import socketio

def on_message(client, userdata, msg):
    topic   = msg.topic          # e.g. "home/sensors/temperature"
    payload = msg.payload.decode("utf-8").strip()
    parts   = topic.split("/")   # ["home", "sensors", "temperature"]

    if len(parts) >= 3 and parts[1] == "sensors":
        sensor_type = parts[2]
        try:
            value = float(payload)
        except ValueError:
            value = payload      # e.g. "DETECTED"

        reading = SensorReading(
            sensor_type=sensor_type,
            value=value,
            timestamp=datetime.utcnow()
        )
        db.session.add(reading)
        db.session.commit()

        # Broadcast to all connected Socket.IO clients
        socketio.emit("sensor_update", {
            "sensor_type": sensor_type,
            "value": value,
            "timestamp": reading.timestamp.isoformat()
        })

        # Trigger anomaly detection for heart-rate readings
        if sensor_type == "heart_rate":
            from app.usecases.health_monitor import check_anomaly
            check_anomaly(value)

    elif len(parts) >= 3 and parts[1] == "status":
        device_code = parts[2]
        socketio.emit("device_status", {
            "device_code": device_code,
            "state": payload
        })

\section{Isolation Forest Inference Module}

\begin{lstlisting}[style=mystyle, language=Python,
  caption={Health anomaly detection --- inference module},
  label={lst:ml}]
import joblib
import numpy as np
from datetime import datetime
from app.domain.entities import Alert
from app.infrastructure.database import db
from app.extensions import socketio

# Load trained model at module import time
model = joblib.load("app/ai/models/isolation_forest.pkl")

RISK_MAP = [
    (130, float("inf"), "emergency"),
    (116, 129,          "warning"),
    (101, 115,          "caution"),
    (50,  100,          "normal"),
    (30,   49,          "caution"),
]

def classify_risk(heart_rate: float) -> str:
    for lo, hi, level in RISK_MAP:
        if lo <= heart_rate <= hi:
            return level
    return "caution"

def build_features(heart_rate: float, ctx: dict) -> np.ndarray:
    now     = datetime.utcnow()
    hour    = now.hour
    is_night = 1 if hour >= 21 or hour < 6 else 0
    light   = ctx.get("light_level", 50.0)
    is_dark  = 1 if light < 50 else 0
    return np.array([[
        heart_rate,
        ctx.get("room_temp",   25.0),
        ctx.get("humidity",    60.0),
        light,
        hour,
        is_night,
        is_dark,
    ]])

def check_anomaly(heart_rate: float, context: dict = None) -> str:
    ctx     = context or {}
    feats   = build_features(heart_rate, ctx)
    score   = model.decision_function(feats)[0]
    pred    = model.predict(feats)[0]   # -1 = anomaly, 1 = normal
    level   = classify_risk(heart_rate)

    # Promote to emergency if IF flags an outlier and level >= caution
    if pred == -1 and level == "caution":
        level = "warning"

    if level != "normal":
        alert = Alert(
            risk_level=level,
            message=f"Heart rate {heart_rate:.0f} bpm detected as {level}",
            timestamp=datetime.utcnow()
        )
        db.session.add(alert)
        db.session.commit()
        socketio.emit("health_alert", {
            "level": level,
            "heart_rate": heart_rate,
            "score": round(float(score), 4),
            "timestamp": alert.timestamp.isoformat()
        })

    return level, float(score)
\end{lstlisting}

\section{Alfred AI Service}

\begin{lstlisting}[style=mystyle, language=Python,
  caption={Alfred AI service --- chat endpoint and context injection},
  label={lst:alfred}]
import ollama
import google.generativeai as genai
from flask import current_app
from app.alfred_knowledge import SYSTEM_PROMPT
from app.usecases.device_usecase import DeviceUseCase

OLLAMA_MODEL = "qwen2.5:3b"

CONTROL_KEYWORDS = {
    "turning on":  "ON",
    "turning off": "OFF",
    "activating":  "ON",
    "switching on":  "ON",
    "switching off": "OFF",
}

def _inject_context(user_msg: str, sensor_ctx: dict) -> str:
    ctx_block = (
        f"[CURRENT SENSOR STATE]\n"
        f"Temperature : {sensor_ctx.get('room_temp', 'N/A')} C\n"
        f"Humidity    : {sensor_ctx.get('humidity', 'N/A')} %\n"
        f"Light Level : {sensor_ctx.get('light_level', 'N/A')}/100\n"
        f"PIR Motion  : {sensor_ctx.get('pir', 'N/A')}\n"
        f"Heart Rate  : {sensor_ctx.get('heart_rate', 'N/A')} bpm "
        f"({sensor_ctx.get('risk_level', 'unknown')} risk)\n"
        f"[END SENSOR STATE]\n\n"
    )
    return ctx_block + user_msg

def _parse_device_command(response_text: str, device_uc: DeviceUseCase):
    text_lower = response_text.lower()
    for phrase, state in CONTROL_KEYWORDS.items():
        if phrase in text_lower:
            for code in ["den", "quat"]:
                if code in text_lower:
                    device_uc.execute_command(code, state)
                    break

def _ollama_chat(enriched_msg: str) -> str:
    response = ollama.chat(
        model=OLLAMA_MODEL,
        messages=[
            {"role": "system", "content": SYSTEM_PROMPT},
            {"role": "user",   "content": enriched_msg},
        ]
    )
    return response["message"]["content"]

def _gemini_fallback(enriched_msg: str) -> str:
    genai.configure(api_key=current_app.config["GEMINI_API_KEY"])
    model = genai.GenerativeModel(
        model_name="gemini-2.5-flash",
        system_instruction=SYSTEM_PROMPT,
    )
    return model.generate_content(enriched_msg).text

def chat(user_msg: str, sensor_ctx: dict,
         device_uc: DeviceUseCase) -> str:
    enriched_msg = _inject_context(user_msg, sensor_ctx)
    try:
        reply = _ollama_chat(enriched_msg)
    except Exception:
        reply = _gemini_fallback(enriched_msg)
    _parse_device_command(reply, device_uc)
    return reply
\end{lstlisting}

monitor via BLE using the \texttt{bleak} library. It scans for the device UUID,

\section{Recommended Production Deployment}

For a production deployment, the following additional steps are recommended:

\begin{enumerate}
    \item \textbf{Reverse proxy}: Place an Nginx reverse proxy in front of the Flask/Gunicorn server to handle static file serving and SSL/TLS termination.
    \item \textbf{TLS certificate}: Obtain a Let's Encrypt certificate using Certbot and configure Nginx to redirect HTTP to HTTPS.
    \item \textbf{Docker}: Containerise the Flask backend and MySQL server using Docker Compose for reproducible deployment and easy horizontal scaling.
    \item \textbf{Process manager}: Use \texttt{systemd} or Docker Compose's \texttt{restart: always} policy to ensure the backend and BLE reader process restart automatically after crashes.
    \item \textbf{Log aggregation}: Configure Flask to write structured JSON logs and forward them to a centralised logging service (e.g., Elastic Stack or Loki) for operational monitoring.
\end{enumerate}

\section{scheduler.py --- Background Job Runner}

\begin{lstlisting}[style=mystyle, language=Python,
  caption={scheduler.py --- APScheduler initialisation and automation evaluator},
  label={lst:scheduler}]
import re
from apscheduler.schedulers.background import BackgroundScheduler
from app.infrastructure.database import db
from app.infrastructure.models import AutomationModel, DeviceStatus, ScheduleModel
from app.gateways.mqtt_publisher import MqttPublisher

_mqtt = MqttPublisher()
_COND_RE = re.compile(
    r"^(value|is_on)\s*(==|!=|>=|<=|>|<)\s*(.+)$"
)

_scheduler = BackgroundScheduler(
    job_defaults={"coalesce": True, "max_instances": 1},
    timezone="Asia/Ho_Chi_Minh",
)

def _eval_condition(auto) -> bool:
    """Evaluate trigger_condition string against live DeviceStatus in DB."""
    m = _COND_RE.match(auto.trigger_condition or "")
    if not m:
        return False
    field, op, thresh = m.groups()
    status = DeviceStatus.query.filter_by(
        device_code=auto.trigger_device_code).first()
    if not status:
        return False
    current = getattr(status, field, None)
    try:
        lhs, rhs = float(current), float(thresh)
        return eval(f"{lhs} {op} {rhs}")  # noqa: S307
    except (TypeError, ValueError):
        return False

def _evaluate_automations():
    """Runs every 30 s. Evaluates all active automation rules."""
    automations = AutomationModel.query.filter_by(is_active=True).all()
    for auto in automations:
        try:
            if _eval_condition(auto):
                _mqtt.publish(
                    f"home/control/{auto.target_device_code}",
                    auto.action_payload,
                )
        except Exception as exc:
            import logging; logging.getLogger(__name__).error(exc)

def _reload_schedules():
    """Runs every 5 min + once at startup. Re-registers cron jobs from DB."""
    existing = {j.id for j in _scheduler.get_jobs()
                if j.id.startswith("sched_")}
    active_ids = set()
    for sched in ScheduleModel.query.filter_by(is_active=True).all():
        job_id = f"sched_{sched.id}"
        active_ids.add(job_id)
        if job_id not in existing:
            _scheduler.add_job(
                _run_schedule, "cron", id=job_id,
                args=[sched.device_id, sched.action],
                **_parse_cron(sched.cron_expr),
            )
    for jid in existing - active_ids:
        _scheduler.remove_job(jid)

def _run_schedule(device_id: int, action: str):
    _mqtt.publish(f"home/control/{device_id}", action)

def _parse_cron(expr: str) -> dict:
    minute, hour, day, month, dow = expr.split()
    return dict(minute=minute, hour=hour,
                day=day, month=month, day_of_week=dow)

def _dispatch_medicine_reminders():
    """Runs every 1 min. Sends due medicine notifications."""
    from app.usecases.medicine_reminder_usecase import dispatch_due_reminders
    dispatch_due_reminders()

def init_scheduler(app):
    """Call from create_app() after db is ready."""
    with app.app_context():
        _scheduler.add_job(_evaluate_automations, "interval",
                           seconds=30, id="automation_eval")
        _scheduler.add_job(_reload_schedules, "interval",
                           minutes=5, id="schedule_sync")
        _scheduler.add_job(_dispatch_medicine_reminders, "interval",
                           minutes=1, id="medicine_dispatch")
        _reload_schedules()   # sync immediately on startup
        _scheduler.start()
\end{lstlisting}

\section{Web Dashboard --- Inline IIFE Voice Module}

The following listing shows the voice control IIFE embedded directly in
\texttt{index.html}. The implementation uses the browser-native
\texttt{SpeechRecognition} API without any external dependencies.

\begin{lstlisting}[style=mystyle, language={[5.0]Lua},
  caption={index.html --- inline IIFE voice control module (JavaScript)},
  label={lst:voice_web}]
<script>
(function () {
  "use strict";

  // Language preference persisted across page reloads
  let lang = localStorage.getItem('voice_lang') || "vi-VN";

  const SpeechRecognition =
    window.SpeechRecognition || window.webkitSpeechRecognition;

  if (!SpeechRecognition) {
    console.warn("Web Speech API not supported in this browser.");
    return;
  }

  const recognition = new SpeechRecognition();
  recognition.continuous    = false;
  recognition.interimResults = false;

  // -- UI Elements ----------------------------------------------------------
  const micBtn   = document.getElementById("voice-mic-btn");
  const langBtn  = document.getElementById("voice-lang-btn");
  const alfredInput = document.getElementById("alfred-win-input");
  // sendAlfredChat() is a module-level function in alfred.js

  function showToast(msg) {
    const t = document.createElement("div");
    t.className = "bat-toast";
    t.textContent = msg;
    document.body.appendChild(t);
    setTimeout(() => t.remove(), 2500);
  }

  // -- Language toggle -------------------------------------------------------
  if (langBtn) {
    langBtn.textContent = lang === "vi-VN" ? "VI" : "EN";
    langBtn.addEventListener("click", () => {
      lang = lang === "vi-VN" ? "en-US" : "vi-VN";
      localStorage.setItem('voice_lang', lang);
      langBtn.textContent = lang === "vi-VN" ? "VI" : "EN";
    });
  }

  // -- Mic button ------------------------------------------------------------
  micBtn.addEventListener("click", () => {
    recognition.lang = lang;
    try {
      recognition.start();
    } catch (e) {
      showToast("Microphone error: " + e.message);
    }
  });

  recognition.onstart = () => {
    micBtn.classList.add("listening");  // red + pulse CSS class
    showToast(lang === "vi-VN" ? "Dang nghe..." : "Listening...");
  };

  recognition.onresult = (event) => {
    const transcript = event.results[0][0].transcript;
    if (alfredInput) {
      alfredInput.value = transcript;
      sendAlfredChat();      // call module-level send function
    }
  };

  recognition.onerror = (event) => {
    const msgs = {
      "not-allowed": "Microphone access denied.",
      "network":     "Network error during recognition.",
      "no-speech":   "No speech detected.",
    };
    showToast(msgs[event.error] || "Voice error: " + event.error);
  };

  recognition.onend = () => {
    micBtn.classList.remove("listening");
  };
})();
</script>
\end{lstlisting}
