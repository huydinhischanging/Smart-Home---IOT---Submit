Full Stack Architecture Diagram (Dùng con lăn chuột để Zoom - Nhấn giữ để Kéo sơ đồ) 
graph TB
    subgraph Clients ["🖥️ Presentation Layer"]
        WEB["Web Browser\n(React/Vite)"]
        MOB["Mobile App\n(Flutter)"]
        ESP["ESP32\n(Embedded C++)"]
    end

    subgraph Proxy_Layer ["🔀 Network & Proxy"]
        Nginx["Nginx Reverse Proxy\n(:80 / :443)"]
        N1["SSL Termination"]
        N2["Static Serving"]
        N3["WS Upgrade"]
        Nginx --- N1 & N2 & N3
    end

    subgraph App_Server ["⚙️ Server (WSGI/ASGI)"]
        Gunicorn["Gunicorn Server"]
        EV["Eventlet Workers\n(Async)"]
        Gunicorn --- EV
    end

    subgraph Core_App ["🐍 Flask Logic Layer"]
        Flask["Flask Application"]
        subgraph Blueprints ["API Endpoints"]
            BP1["/auth (JWT)"]
            BP2["/devices (CRUD)"]
            BP3["/health (HRV)"]
            BP4["/ai (Gemini)"]
        end
        SIO["Socket.IO Server\n(Real-time)"]
        Flask --- Blueprints & SIO
    end

    subgraph Backend_Services ["🔧 Persistence"]
        MQTT_C["MQTT Client\n(Paho)"]
        SCHED["APScheduler"]
        DB[(PostgreSQL)]
    end

    subgraph Broker_Layer ["📡 Message Broker"]
        Broker["EMQX Broker\n(:1883)"]
        T1["control/#"]
        T2["status/#"]
        T3["sensors/#"]
        Broker --- T1 & T2 & T3
    end

    %% Connections
    WEB & MOB -->|"HTTPS REST + WSS"| Nginx
    Nginx --> Gunicorn
    Gunicorn --> Flask
    Flask --> DB
    BP2 -->|"Cmd"| MQTT_C
    MQTT_C --> Broker
    Broker --> ESP
    MQTT_C & SCHED -->|"Emit"| SIO
    SIO -.->|"Push"| WEB & MOB
    ESP --> Broker
