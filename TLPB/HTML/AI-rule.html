Architecture: Context-Aware Smart Home System

flowchart TD
    U["User message"] --> API["POST /api/ai/chat"]
    API --> HC["AIUseCase.handle_chat"]

    %% NEW: DB context retrieval (Data Enrichment Layer)
    subgraph Context_Layer [Context Retrieval Layer]
        HC --> CTX["Load realtime house context"]
        CTX --> DBR[("Read DB: devices, status, rooms, floors, user scope")]
        DBR --> CACHE["Optional cache (Redis/TTL)"]
    end

    %% Decision after context
    CACHE --> R{"Rule matched with context?"}

    %% PATH 1: Device control (Operational Path)
    subgraph IoT_Control_Path [IoT Operational Path]
        R -->|Yes| PLAN["Create action plan"]
        PLAN --> EXE["_execute_plan"]
        EXE --> CTRL["device_usecase.control_device"]
        CTRL --> CMD["send_command"]
        CMD --> MQTT_PUB["Publish home/control/<device_code>"]
    end

    MQTT_PUB --> ESP["ESP32 executes"]
    ESP --> ST_PUB["Publish home/status/<device_code>"]
    ST_PUB --> LISTENER["Backend MQTT listener"]
    
    LISTENER --> DBW[("Write status/log to DB")]
    LISTENER --> EMIT["Socket.IO emit device_status"]
    EMIT --> UI["Web + Mobile sync"]

    %% PATH 2: Conversational response (AI Path)
    subgraph AI_Conversational_Path [AI Logic Path]
        R -->|No| MODE{"AI mode"}
        MODE -->|auto/llm| OLLAMA["Local LLM Ollama"]
        OLLAMA -->|success| RESP["Text response"]
        OLLAMA -->|fail/timeout| GEMINI["Cloud Gemini fallback"]
        MODE -->|gemini| GEMINI
        MODE -->|rule-only| RULE_RESP["Rule text response"]
        RULE_RESP --> RESP
        GEMINI --> RESP
    end

    RESP --> UI
