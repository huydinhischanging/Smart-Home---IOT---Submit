FIRMWARE STATE MACHINE
stateDiagram-v2
    [*] --> BOOT

    state BOOT {
        [*] --> init_serial
        init_serial --> init_gpio
        init_gpio --> load_config
        load_config --> [*]
        note right of load_config
            GPIO17: Relay Den | GPIO16: Relay Quat
            GPIO32: Light Sensor (ADC)
        end note
    }

    BOOT --> WIFI_CONNECTING

    state WIFI_CONNECTING {
        [*] --> wifi_begin
        wifi_begin --> wait_connected
        wait_connected --> wait_connected : timeout < 20s
        wait_connected --> WIFI_OK : status == WL_CONNECTED
        wait_connected --> WIFI_FAIL : timeout >= 20s
    }

    WIFI_FAIL --> WIFI_CONNECTING : retry 5s
    WIFI_OK --> MQTT_CONNECTING

    state MQTT_CONNECTING {
        [*] --> set_broker
        set_broker --> set_callback
        set_callback --> client_connect
        client_connect --> MQTT_OK : connected
        client_connect --> MQTT_FAIL : refused
    }

    MQTT_FAIL --> MQTT_CONNECTING : retry 3s
    MQTT_OK --> RUNNING

    state RUNNING {
        [*] --> main_loop
        state main_loop {
            mqtt_loop --> read_sensor
            read_sensor --> publish_sensor
            publish_sensor --> check_status
            check_status --> mqtt_loop
        }
        
        state MQTT_CALLBACK {
            parse_topic --> relay_ACTION
            relay_ACTION --> publish_ack
        }
        
        main_loop --> MQTT_CALLBACK : Message Arrived
        MQTT_CALLBACK --> main_loop : Done
    }

    RUNNING --> WIFI_CONNECTING : WiFi Lost
    RUNNING --> MQTT_CONNECTING : MQTT Lost
