    USERS {
        int id PK
        string username
        string email
        string password
        enum role
        bool is_active
        datetime created_at
    }

    ROOMS {
        int id PK
        int user_id FK
        string name
        json polygon_data
        string floorplan_url
    }

    DEVICE {
        int id PK
        int user_id FK
        int room_id FK
        string name
        string code "Unique per user"
        json control_types
        string status
        bool is_active
    }

    DEVICE_STATUS {
        int id PK
        int device_id FK
        string status
        float rssi
        datetime recorded_at
    }

    SENSOR_DATA {
        int id PK
        int device_id FK
        float value
        datetime created_at
    }

    CONTROL_LOG {
        int id PK
        int device_id FK
        int user_id FK
        string action
        string source
        datetime created_at
    }

    ALERTS {
        int id PK
        int user_id FK
        int device_id FK
        string message
        string severity
        bool is_read
    }

    NOTIFICATIONS {
        int id PK
        int user_id FK
        int alert_id FK
        string title
        string body
        bool is_read
    }

    PATIENT_PROFILES {
        int id PK
        int user_id FK
        string patient_name
        int age
        float baseline_hr_rest
        text medications
    }

    PATIENT_HR_RECORDS {
        int id PK
        int user_id FK
        int bpm
        float hrv_rmssd
        float hrv_sdnn
        string hrv_risk
        datetime recorded_at
    }

    MEDICINE_REMINDERS {
        int id PK
        int user_id FK
        string medicine_name
        string time_of_day
        enum recurrence
        bool is_active
    }
